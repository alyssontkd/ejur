    ALGORITHM = UNDEFINED 
    DEFINER = `root`@`localhost` 
    SQL SECURITY DEFINER
VIEW `bdejur`.`auth` AS
    (SELECT 
        `bdejur`.`login`.`id_usuario` AS `id_usuario`,
        `bdejur`.`perfil`.`id_perfil` AS `id_perfil`,
        `bdejur`.`email`.`em_email` AS `em_email`,
        `bdejur`.`login`.`pw_senha` AS `pw_senha`,
        `bdejur`.`usuario`.`nm_usuario` AS `nm_usuario`,
        1 AS `id_contrato`
    FROM
        (((`bdejur`.`usuario`
        JOIN `bdejur`.`login` ON ((`bdejur`.`login`.`id_usuario` = `bdejur`.`usuario`.`id_usuario`)))
        JOIN `bdejur`.`email` ON ((`bdejur`.`email`.`id_email` = `bdejur`.`login`.`id_email`)))
        JOIN `bdejur`.`perfil` ON ((`bdejur`.`perfil`.`id_perfil` = `bdejur`.`login`.`id_perfil`))));
